const getTopStates = (limit: number = 5): [string, number][] => {
  const stateCounts = contacts.reduce((acc, contact) => {
    const state = contact.state || ''
    acc[state] = (acc[state] || 0) + 1
    return acc
  }, {} as Record<string, number>)

  return Object.entries(stateCounts)
    .sort(([, a], [, b]) => b - a)
