```python
def validate_phone(phone_str):
    """
    Valida e normaliza telefone brasileiro
    
    Aceita:
    - 11999999999
    - (11) 99999-9999
    - +55 11 99999-9999
    - 5511999999999
    
    Retorna: +5511999999999 (E.164)
    """
    import re
    
    # Remover formata√ß√£o
    clean = re.sub(r'[^\d+]', '', phone_str)
    
    # Remover zeros √† esquerda (exceto c√≥digo do pa√≠s)
    if clean.startswith('0'):
        clean = clean.lstrip('0')
    
    # Adicionar +55 se n√£o tiver c√≥digo de pa√≠s
    if not clean.startswith('+'):
        if clean.startswith('55'):
            clean = f'+{clean}'
        else:
            clean = f'+55{clean}'
    
    # Validar comprimento (Brasil: +55 + DDD(2) + N√∫mero(8-9))
    # Total: 13-14 d√≠gitos com +
    if len(clean) < 13 or len(clean) > 14:
        raise ValidationError(
            f'Telefone inv√°lido: {phone_str}. '
            'Deve ter 10-11 d√≠gitos (DDD + n√∫mero)'
        )
    
    # Validar DDD (11-99)
    ddd = clean[3:5]  # Ap√≥s +55
    if not (11 <= int(ddd) <= 99):
        raise ValidationError(f'DDD inv√°lido: {ddd}')
    
    return clean
```

```python
    """
    Valida data em m√∫ltiplos formatos
```
